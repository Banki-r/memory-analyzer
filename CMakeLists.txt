    cases/matchers/UninitializedMatcher.cpp
    cases/matchers/ContainerMatcher.cpp
    cases/matchers/LinkedlistImplMatcher.cpp

    cases/matchers/IMatcher.h
    cases/visitors/IASTVisitor.h

    view/MainWindow.h
    view/MainWindow.cpp
